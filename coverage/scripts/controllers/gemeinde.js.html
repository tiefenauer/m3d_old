    };
 
    
    GemeindeController.prototype.tab = 'search';
 
    GemeindeController.prototype.init = function(){
      $.ajax({
        url: 'assets/gemeinden/gemeinden.json',
        success: function(data){
          $scope.gemeinden.all = data;

          $scope.$apply();
